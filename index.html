Please try again."); }); } // Auto-focus and auto-submit for OTP document.getElementById('otpInput').addEventListener('input', function(e) { this.value = this.value.replace(/[^0-9]/g, ''); if (this.value.length === 6) { setTimeout(() => submitOTP(), 500); } });
